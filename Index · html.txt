/* ════════════════════════════════════════
   LOAD / SAVE
   ════════════════════════════════════════ */
function load() {
  try {
    const s = localStorage.getItem(KEY.settings);
    if (s) settings = { ...settings, ...JSON.parse(s) };
    const h = localStorage.getItem(KEY.history);
    if (h) history = JSON.parse(h);
    const st = localStorage.getItem(KEY.stats);
    if (st) stats = { ...stats, ...JSON.parse(st) };
    const pr = localStorage.getItem(KEY.progress);
    if (pr) {
      const p = JSON.parse(pr);
      const today = new Date().toDateString();
      state.progress = p.date === today ? p.count : 0;
    }
  } catch(e) {}
}

function save() {
  try {
    localStorage.setItem(KEY.settings, JSON.stringify(settings));
    localStorage.setItem(KEY.history,  JSON.stringify(history));
    localStorage.setItem(KEY.stats,    JSON.stringify(stats));
    localStorage.setItem(KEY.progress, JSON.stringify({
      date: new Date().toDateString(),
      count: state.progress,
    }));
  } catch(e) {}
}

function navigate(screenId) {
  document.querySelectorAll('.screen').forEach(s => s.classList.remove('active'));
  document.getElementById(`screen-${screenId}`).classList.add('active');
  document.querySelectorAll('.nav-btn').forEach(b => {
    b.classList.toggle('active', b.dataset.screen === screenId);
  });
  if (screenId === 'history')  renderHistory();
  if (screenId === 'settings') renderSettings();
}

   SCENE TABS
   ════════════════════════════════════════ */
function buildSceneTabs(containerId, onChange) {
  const el = document.getElementById(containerId);
  el.innerHTML = '';
  SCENES.forEach(sc => {
    const btn = document.createElement('button');
    btn.className = `tag ${sc.id === state.scene ? 'tag-active' : 'tag-default'}`;
    btn.textContent = `${sc.icon} ${sc.label}`;
    btn.onclick = () => {
      state.scene = sc.id;
      document.querySelectorAll(`#${containerId} .tag`).forEach(t => {
        t.className = `tag ${t.textContent.includes(sc.label) ? 'tag-active' : 'tag-default'}`;
      });
      // Sync other tab bar
      ['sceneTabs','sceneTabsPrac'].forEach(id => {
        if (id !== containerId) {
          document.querySelectorAll(`#${id} .tag`).forEach(t => {
            const active = SCENES.find(s => t.textContent.includes(s.label))?.id === sc.id;
            t.className = `tag ${active ? 'tag-active' : 'tag-default'}`;
          });
        }
      });
      document.getElementById('homeBadge').textContent = `${sc.icon} ${sc.label}`;
      if (onChange) onChange(sc.id);
    };
    el.appendChild(btn);
  });
}

function toggleMic() {
  if (!('webkitSpeechRecognition' in window) && !('SpeechRecognition' in window)) {
    showToast('このブラウザは音声認識に対応していません');
    return;
  }
  const btn = document.getElementById('micBtn');
  const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;
  const rec = new SpeechRecognition();
  rec.lang = 'ja-JP';
  rec.interimResults = false;
  rec.onstart = () => btn.classList.add('listening');
  rec.onresult = e => {
    const text = e.results[0][0].transcript;
    state.inputText = text;
    document.getElementById('micText').textContent = `🎤 "${text}"`;
    document.getElementById('micText').className = 'mic-text has-input';
  };
  rec.onend  = () => btn.classList.remove('listening');
  rec.onerror = () => { btn.classList.remove('listening'); showToast('音声認識に失敗しました'); };
  rec.start();
}

async function doTranslate() {
  const text = state.inputText;
  if (!text) { showToast('フレーズを選ぶかマイクで話してください'); return; }

  const btn = document.getElementById('translateBtn');
  btn.innerHTML = '<div class="spinner"></div> 変換中…';
  btn.disabled = true;

  try {
    const en = await translateWithAI(text);
    state.resultEn = en;
    state.resultJp = text;

    document.getElementById('resultEn').textContent = en;
    document.getElementById('resultJp').textContent = settings.showJp ? `日本語: ${text}` : '';
    document.getElementById('resultArea').style.display = 'block';
    document.getElementById('partnerSection').style.display = 'none';

    // Auto speak
    if (settings.autoSpeak) tts(en);

    // Pronunciation score (simulated)
    if (settings.showScore) showScore();

    // Update progress
    incrementProgress();

    // Update partner who label
    const char = CHARACTERS.find(c => c.id === settings.character) || CHARACTERS[0];
    document.getElementById('partnerWho').textContent = `${char.avatar} ${char.name} の返答`;

  } catch(e) {
    showToast('翻訳に失敗しました。もう一度お試しください。');
  }

  btn.innerHTML = '<span>✨</span> 英語に変換する';
  btn.disabled = false;
}

async function translateWithAI(japaneseText) {
  const scene = SCENES.find(s => s.id === state.scene);
  const sceneName = scene ? scene.label : '一般';

  // Check if it matches a known phrase first
  const phrases = PHRASES[state.scene] || [];
  const known = phrases.find(p => p.jp === japaneseText || p.en === japaneseText);
  if (known) return known.en;

  // Call Anthropic API
  const resp = await fetch('https://api.anthropic.com/v1/messages', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      model: 'claude-sonnet-4-6',
      max_tokens: 200,
      messages: [{
        role: 'user',
        content: `シーン: ${sceneName}。次の日本語を自然な旅行英会話の英語1文に翻訳してください。英語のみ答えてください。日本語: ${japaneseText}`,
      }],
    }),
  });
  const data = await resp.json();
  const txt = data.content?.map(b => b.text || '').join('').trim();
  return txt || japaneseText;
}

function repeatPractice() {
  if (!state.resultEn) return;
  let count = 0;
  const interval = setInterval(() => {
    tts(state.resultEn);
    count++;
    if (count >= 3) clearInterval(interval);
  }, 2000);
  showToast('3回リピート開始！');
}

async function getPartnerReply() {
  const sec = document.getElementById('partnerSection');
  const dots = document.getElementById('typingDots');
  const enEl = document.getElementById('partnerEnText');
  const jpEl = document.getElementById('partnerJp');

  sec.style.display = 'block';
  dots.style.display = 'flex';
  enEl.textContent = '';
  jpEl.textContent = '';

  // Use pre-defined reply bank first
  const bank = PARTNER_REPLIES[state.scene] || PARTNER_REPLIES.greeting;
  const reply = bank[Math.floor(Math.random() * bank.length)];

  try {
    // Try AI-generated reply if we have input
    if (state.resultEn) {
      const aiReply = await getAIReply(state.resultEn);
      dots.style.display = 'none';
      enEl.textContent = aiReply.en;
      jpEl.textContent = settings.showJp ? aiReply.jp : '';
      tts(aiReply.en);
      return;
    }
  } catch(e) {}

  dots.style.display = 'none';
  enEl.textContent = reply.en;
  jpEl.textContent = settings.showJp ? reply.jp : '';
  tts(reply.en);
}

async function getAIReply(userEn) {
  const char = CHARACTERS.find(c => c.id === settings.character) || CHARACTERS[0];
  const resp = await fetch('https://api.anthropic.com/v1/messages', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      model: 'claude-sonnet-4-6',
      max_tokens: 200,
      messages: [{
        role: 'user',
        content: `You are ${char.name}, a ${char.role}. Respond naturally in English to: "${userEn}". Keep reply to 1-2 sentences. Then on a new line write the Japanese translation starting with "JP:". Format: ENGLISH_REPLY\nJP: JAPANESE_TRANSLATION`,
      }],
    }),
  });
  const data = await resp.json();
  const text = data.content?.map(b => b.text || '').join('').trim() || '';
  const lines = text.split('\n');
  const en = lines[0] || reply.en;
  const jp = (lines.find(l => l.startsWith('JP:')) || '').replace('JP:', '').trim();
  return { en, jp };
}

function showScore() {
  const score = Math.floor(60 + Math.random() * 40); // 60–99
  const card = document.getElementById('scoreCard');
  const ring = document.getElementById('scoreRing');
  const num  = document.getElementById('scoreNum');
  const fb   = document.getElementById('scoreFeedback');

  card.style.display = 'block';
  num.textContent = score;

  const pct = score / 100;
  ring.style.strokeDashoffset = 188.5 * (1 - pct);
  ring.style.stroke = score >= 80 ? 'var(--green)' : score >= 60 ? 'var(--amber)' : 'var(--rose)';

  fb.textContent = score >= 85 ? '🎉 素晴らしい発音です！'
    : score >= 70 ? '👍 なかなか上手！もう少し練習しよう'
    : '💪 もう一度練習してみましょう';

  stats.scores.push(score);
  if (stats.scores.length > 100) stats.scores.shift();
  save();
}

function incrementProgress() {
  if (state.progress < settings.dailyGoal) {
    state.progress++;
    stats.total++;
    if (state.progress === 1) stats.days++;

    // Add to history
    history.unshift({
      en   : state.resultEn,
      jp   : state.resultJp,
      scene: state.scene,
      score: settings.showScore ? Math.floor(60 + Math.random() * 40) : null,
      time : Date.now(),
    });
    if (history.length > 200) history.pop();

    save();
    renderHome();
    updatePracBadge();

    if (state.progress === settings.dailyGoal) {
      setTimeout(() => document.getElementById('completeOverlay').classList.add('show'), 500);
    }
  }
}

function updatePracBadge() {
  document.getElementById('pracBadge').textContent = `${state.progress} / ${settings.dailyGoal}`;
}

/* ════════════════════════════════════════
   HISTORY SCREEN
   ════════════════════════════════════════ */
function renderHistory() {
  // Heatmap (90 days)
  const hm = document.getElementById('heatmap');
  hm.innerHTML = '';
  const activeDays = new Set(history.map(h => new Date(h.time).toDateString()));
  for (let i = 89; i >= 0; i--) {
    const d = new Date();
    d.setDate(d.getDate() - i);
    const key = d.toDateString();
    const count = history.filter(h => new Date(h.time).toDateString() === key).length;
    const cell = document.createElement('div');
    cell.className = `heatmap-cell${count >= 10 ? ' d3' : count >= 5 ? ' d2' : count >= 1 ? ' d1' : ''}`;
    cell.title = `${key}: ${count}フレーズ`;
    hm.appendChild(cell);
  }

  // History list
  const list = document.getElementById('historyList');
  list.innerHTML = '';
  if (!history.length) {
    list.innerHTML = `<div style="text-align:center;color:var(--muted);padding:40px 0;font-size:14px">まだ練習履歴がありません<br>練習画面で英語を話してみよう！</div>`;
    return;
  }
  history.slice(0, 30).forEach(h => {
    const scene = SCENES.find(s => s.id === h.scene);
    const item = document.createElement('div');
    item.className = 'history-item';
    item.innerHTML = `
      <div class="history-icon">${scene?.icon || '💬'}</div>
      <div>
        <div class="history-en">${h.en}</div>
        <div class="history-jp">${h.jp}</div>
        <div class="history-meta">
          <span class="history-tag history-tag-scene">${scene?.label || '—'}</span>
          ${h.score ? `<span class="history-tag history-tag-score">発音 ${h.score}点</span>` : ''}
        </div>
      </div>`;
    list.appendChild(item);
  });
}

/* ════════════════════════════════════════
   TTS
   ════════════════════════════════════════ */
function tts(text, lang = 'en-US') {
  if (!window.speechSynthesis) return;
  window.speechSynthesis.cancel();
  const char = CHARACTERS.find(c => c.id === settings.character);
  const utt = new SpeechSynthesisUtterance(text);
  utt.lang = char?.voice || lang;
  utt.rate = 0.9;
  utt.pitch = 1;
  window.speechSynthesis.speak(utt);
}

/* ════════════════════════════════════════
   PWA: SERVICE WORKER + INSTALL
   ════════════════════════════════════════ */
if ('serviceWorker' in navigator) {
  // Inline SW blob
  const swCode = `
    const CACHE = 'talkgo-v1';
    self.addEventListener('install', e => {
      self.skipWaiting();
    });
    self.addEventListener('activate', e => {
      e.waitUntil(caches.keys().then(keys =>
        Promise.all(keys.filter(k => k !== CACHE).map(k => caches.delete(k)))
      ));
    });
    self.addEventListener('fetch', e => {
      if (e.request.url.includes('api.anthropic.com') ||
          e.request.url.includes('fonts.googleapis.com')) return;
      e.respondWith(
        caches.match(e.request).then(cached => {
          if (cached) return cached;
          return fetch(e.request).then(res => {
            const clone = res.clone();
            caches.open(CACHE).then(c => c.put(e.request, clone));
            return res;
          }).catch(() => cached);
        })
      );
    });
  `;
  const blob = new Blob([swCode], { type: 'text/javascript' });
  const swUrl = URL.createObjectURL(blob);
  navigator.serviceWorker.register(swUrl).catch(() => {});
}

window.addEventListener('beforeinstallprompt', e => {
  e.preventDefault();
  deferredInstall = e;
  document.getElementById('installBanner').style.display = 'flex';
});

function doInstall() {
  if (!deferredInstall) {
    showToast('インストール: ブラウザメニュー →「ホーム画面に追加」');
    return;
  }
  deferredInstall.prompt();
  deferredInstall.userChoice.then(r => {
    if (r.outcome === 'accepted') {
      showToast('🎉 インストールしました！');
      deferredInstall = null;
      document.getElementById('installBanner').style.display = 'none';
    }
  });
}
